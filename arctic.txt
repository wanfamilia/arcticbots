#ALIAS {log}
{
	#if {"%0" == "on"}
	{
		#if {"$logging" != "on"}
		{
			#format {time} {%T};
			#format {logfile} {logs/%d.txt} {$time};
			#log append $logfile;
			#variable logging on
		}
		{
			#showme You are already logging.
		}
	};
	#if {"%0" == "off"}
	{
		#if {"$logging" == "on"}
		{
			#log;
			#variable logging off
		}
		{
			#showme You are not logging.
		}
	};
	#if {"%0" != "off" && "%0" != "on"}
	{
		#showme Turn log on or off?
	}
}
